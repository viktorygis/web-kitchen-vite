Динамические компоненты в React

Динамические компоненты — это компоненты, которые выбираются и отображаются в зависимости от условий или данных.
Это позволяет строить гибкие интерфейсы, где контент меняется "на лету".

Основы динамических компонентов


function Page({type}) {
  if (type === 'main') return <Main />;
  if (type === 'about') return <About />;
  return <NotFound />;
}
В зависимости от type выбирается нужный компонент.

Условный рендеринг компонентов


function Status({online}) {
  return online
    ? <span style={{color:"green"}}>Онлайн</span>
    : <span style={{color:"gray"}}>Оффлайн</span>;
}
Можно использовать if , switch , тернарные операторы и даже массивы для динамического выбора.

Рендеринг компонента из переменной


const components = {
  main: Main,
  about: About,
  help: Help
};
function Page({type}) {
  const Tag = components[type] || NotFound;
  return <Tag />;
}
Можно хранить компоненты в объекте и выбирать их по ключу!

Практика: динамическая панель


<div id="root"></div>
<script type="text/babel">
function Home() { return <div>🏠 Домашняя страница</div>; }
function Profile() { return <div>👤 Профиль</div>; }
function Settings() { return <div>⚙ Настройки</div>; }

const components = { home: Home, profile: Profile, settings: Settings };

function App() {
  const [active, setActive] = React.useState('home');
  const ActiveComponent = components[active];
  return (
    <div>
      <nav>
        {Object.keys(components).map(key =>
          <button key={key} onClick={() => setActive(key)}>{key}</button>
        )}
      </nav>
      <ActiveComponent />
    </div>
  );
}
const root = ReactDOM.createRoot(document.getElementById('root'));
root.render(<App />);
</script>

Советы и типичные ошибки

  • Не передавайте компонент как строку: используйте переменные!
  • Компонент с большой буквы — это функция, а не строка.
  • Проверяйте, что динамический компонент существует.

Лучшие практики

  • Объединяйте динамические компоненты в объект или Map для удобного доступа.
  • Избегайте лишних условий — используйте ключи и переменные.
  • Документируйте, какие значения можно передавать для динамического выбора.

Примеры динамического рендеринга

Switch-case для разных видов


function RenderByType({type}) {
  switch(type) {
    case 'a': return <A />;
    case 'b': return <B />;
    default: return <NotFound />;
  }
}

Массив компонентов


const steps = [StepOne, StepTwo, StepThree];
function Wizard({step}) {
  const Step = steps[step] || NotFound;
  return <Step />;
}
