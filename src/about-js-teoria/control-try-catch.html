Обработка ошибок: Try Catch Finally

Try Catch Finally — это механизм обработки ошибок (исключений) в JavaScript. Позволяет перехватывать ошибки, обрабатывать их и продолжать выполнение программы без её краха.

Основные конструкции

  • try — блок кода, где могут возникнуть ошибки
  • catch — блок обработки ошибок, если они произойдут
  • finally — блок кода, который выполнится в любом случае
  • throw — оператор для генерации ошибок вручную

Try Catch: перехват ошибок

Try блок содержит код, который может вызвать ошибку. Catch блок выполняется, если в try возникнет ошибка.

Синтаксис:

try {
  // Код, который может вызвать ошибку
  riskyOperation();
} catch (error) {
  // Код обработки ошибки
  console.log("Ошибка перехвачена:", error.message);
}

Примеры Try Catch:

// Пример 1: Перехват ошибки доступа к свойству
try {
  const obj = null;
  console.log(obj.property); // Ошибка!
} catch (error) {
  console.log("Поймана ошибка:", error.message);
  // Вывод: "Поймана ошибка: Cannot read property 'property' of null"
}

// Пример 2: Перехват ошибки парсинга JSON
try {
  const json = "{invalid json}";
  const data = JSON.parse(json); // Ошибка!
} catch (error) {
  console.log("JSON не валиден:", error.message);
  // Вывод: "JSON не валиден: Unexpected token i in JSON at position 1"
}

// Пример 3: Продолжение выполнения после ошибки
try {
  const result = JSON.parse('{"name": "John"}');
  console.log("Успешно:", result);
} catch (error) {
  console.log("Ошибка парсинга");
}
console.log("Программа продолжает работу"); // Выполнится в любом случае

Объект Error

try {
  // Генерируем ошибку
  throw new Error("Что-то пошло не так!");
} catch (error) {
  // Свойства объекта ошибки
  console.log("Сообщение:", error.message);   // "Что-то пошло не так!"
  console.log("Тип:", error.name);            // "Error"
  console.log("Stack trace:", error.stack);   // Полная информация об ошибке
}

Finally: код выполняется всегда

Finally блок выполняется независимо от того, произошла ошибка или нет. Используется для очистки ресурсов.

Синтаксис:

try {
  // Рискованный код
} catch (error) {
  // Обработка ошибки
} finally {
  // Выполнится в любом случае
}

Примеры Finally:

// Пример 1: Закрытие ресурсов
let resource;
try {
  console.log("1. Открываем ресурс");
  resource = "открыт";
  throw new Error("Ошибка при работе");
} catch (error) {
  console.log("2. Обработка ошибки:", error.message);
} finally {
  console.log("3. Закрываем ресурс:", resource);
}
// Вывод:
// 1. Открываем ресурс
// 2. Обработка ошибки: Ошибка при работе
// 3. Закрываем ресурс: открыт

// Пример 2: Логирование
try {
  const result = 10 / 2;
  console.log("Результат:", result);
} finally {
  console.log("Операция завершена");
}
// Вывод:
// Результат: 5
// Операция завершена

// Пример 3: Try без catch (но с finally)
try {
  console.log("Выполнение кода");
} finally {
  console.log("Очистка ресурсов");
}

// Пример 4: Return в try и finally
function test() {
  try {
    console.log("В try");
    return "результат из try";
  } finally {
    console.log("В finally"); // Выполнится ДО return
  }
}

console.log(test());
// Вывод:
// В try
// В finally
// результат из try

Когда использовать finally

  • Закрытие файлов и соединений с БД
  • Освобождение памяти и ресурсов
  • Логирование и отслеживание
  • Очистка глобального состояния

Типы ошибок в JavaScript

JavaScript имеет встроенные типы ошибок для разных ситуаций. Каждый тип расширяет базовый класс Error.

Тип ошибки Описание Пример
Error Базовый тип ошибки throw new Error("Ошибка")
SyntaxError Ошибка синтаксиса eval("!!!")
TypeError Ошибка типа null.method()
ReferenceError Ошибка ссылки на переменную console.log(undefinedVar)
RangeError Ошибка диапазона new Array(-1)
URIError Ошибка URI функции decodeURIComponent("%")

Перехват различных типов ошибок:

try {
  // Какая-то операция
  riskyFunction();
} catch (error) {
  if (error instanceof TypeError) {
    console.log("Ошибка типа:", error.message);
  } else if (error instanceof ReferenceError) {
    console.log("Ошибка ссылки:", error.message);
  } else if (error instanceof Error) {
    console.log("Неизвестная ошибка:", error.message);
  }
}

Оператор throw: генерация ошибок

Throw генерирует ошибку вручную. Используется для создания пользовательских ошибок и валидации.

Синтаксис:

throw new Error("Описание ошибки");
throw new TypeError("Неправильный тип");
throw "Просто строка"; // Можно бросить что угодно

Примеры throw:

// Пример 1: Валидация параметров
function divide(a, b) {
  if (typeof a !== "number" || typeof b !== "number") {
    throw new TypeError("Оба параметра должны быть числами");
  }
  if (b === 0) {
    throw new Error("Деление на ноль невозможно");
  }
  return a / b;
}

try {
  console.log(divide(10, 2));    // 5
  console.log(divide(10, 0));    // Ошибка!
} catch (error) {
  console.log("Ошибка:", error.message);
}

// Пример 2: Пользовательский класс ошибки
class ValidationError extends Error {
  constructor(message) {
    super(message);
    this.name = "ValidationError";
  }
}

function validateAge(age) {
  if (age < 0 || age > 120) {
    throw new ValidationError("Возраст должен быть между 0 и 120");
  }
  return true;
}

try {
  validateAge(-5);
} catch (error) {
  if (error instanceof ValidationError) {
    console.log("Ошибка валидации:", error.message);
  }
}

// Пример 3: Цепочка обработки
function processData(data) {
  try {
    if (!data) {
      throw new Error("Данные не предоставлены");
    }
    const parsed = JSON.parse(data);
    if (!parsed.name) {
      throw new Error("Поле 'name' не найдено");
    }
    return parsed;
  } catch (error) {
    console.log("Ошибка обработки:", error.message);
    throw error; // Пробрасываем ошибку дальше
  }
}

Практические примеры

Пример 1: Безопасный парсинг JSON

function parseJSON(jsonString) {
  try {
    const data = JSON.parse(jsonString);
    return { success: true, data };
  } catch (error) {
    return { success: false, error: error.message };
  }
}

console.log(parseJSON('{"name": "John"}'));
// { success: true, data: { name: 'John' } }

console.log(parseJSON("invalid"));
// { success: false, error: "Unexpected token i in JSON at position 0" }

Пример 2: Валидация форм

function validateForm(formData) {
  const errors = [];

  try {
    if (!formData.email) {
      throw new Error("Email обязателен");
    }
    if (!formData.email.includes("@")) {
      throw new Error("Email невалиден");
    }
    if (!formData.password || formData.password.length < 8) {
      throw new Error("Пароль должен быть не менее 8 символов");
    }
    return { valid: true };
  } catch (error) {
    errors.push(error.message);
    return { valid: false, errors };
  }
}

const result = validateForm({ email: "test", password: "123" });
console.log(result);

Пример 3: Работа с асинхронным кодом

// async/await с try-catch
async function fetchUserData(userId) {
  let userData;
  try {
    console.log("Загружаем данные...");
    const response = await fetch(`/api/users/${userId}`);

    if (!response.ok) {
      throw new Error(`HTTP ${response.status}`);
    }

    userData = await response.json();
    return userData;
  } catch (error) {
    console.log("Ошибка при загрузке:", error.message);
    return null;
  } finally {
    console.log("Загрузка завершена");
  }
}

Ловушки и рекомендации

⚠️ Основные ловушки

  • Пустой catch блок — скрывает ошибки. Всегда логируйте или обрабатывайте ошибку
  • Catch перехватывает ВСЕ ошибки — включая ошибки в коде самого catch блока
  • Try-catch не ловит асинхронные ошибки — используйте async/await или .catch()
  • Finally с return — finally выполняется перед return, но может переопределить значение
  • Производительность — try-catch добавляет небольшой оверхед, не используйте для управления потоком

✅ Лучшие практики

  • Обрабатывайте ошибки конкретным образом, не игнорируйте их
  • Используйте специфические типы ошибок (TypeError, ValidationError и т.д.)
  • Логируйте ошибки для отладки и мониторинга
  • Используйте finally для очистки ресурсов
  • Для асинхронного кода используйте async/await с try-catch
  • Не используйте try-catch для управления потоком выполнения
  • Пробрасывайте ошибки вверх, если не знаете, как их обработать

Плохо: пустой catch блок

// ❌ Плохо - ошибка скрыта
try {
  riskyOperation();
} catch (error) {
  // Ничего не делаем - ошибка потеряна!
}

Хорошо: обработка ошибки

// ✅ Хорошо - ошибка обработана
try {
  riskyOperation();
} catch (error) {
  console.error("Произошла ошибка:", error);
  // Или отправить на сервер логирования
  logErrorToServer(error);
}
