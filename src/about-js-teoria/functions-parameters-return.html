Параметры и Return в функциях

Параметры — это значения, которые функция получает для работы. Return — это значение, которое функция возвращает. Вместе они образуют интерфейс функции для взаимодействия с внешним кодом.

Основные концепции

  • Параметры — переменные в определении функции
  • Аргументы — значения, передаваемые при вызове функции
  • Return — возвращаемое значение функции
  • Параметры по умолчанию — значения, если аргумент не передан
  • Rest параметр — захват всех аргументов в массив
  • Spread оператор — раскрытие массива в аргументы

Параметры функции

Параметры — это переменные, объявленные в скобках при определении функции. Они получают значения (аргументы) при вызове функции.

Синтаксис:

// Определение функции с параметрами
function greet(name, greeting) {
  console.log(greeting + ", " + name);
}

// Вызов функции с аргументами
greet("Alice", "Hello");  // "Hello, Alice"
greet("Bob", "Hi");       // "Hi, Bob"

Примеры с параметрами:

// Пример 1: Функция с несколькими параметрами
function add(a, b, c) {
  return a + b + c;
}

console.log(add(1, 2, 3));      // 6
console.log(add(10, 20, 30));   // 60

// Пример 2: Параметры это локальные переменные
function printParams(x, y) {
  console.log("x =", x);
  console.log("y =", y);
  x = 100; // Изменение параметра не влияет на внешнюю переменную
}

let a = 5, b = 10;
printParams(a, b);
console.log("a =", a); // 5 (не изменилось)

// Пример 3: Функция без параметров
function sayHello() {
  return "Привет, мир!";
}

console.log(sayHello()); // "Привет, мир!"

// Пример 4: Больше аргументов, чем параметров (лишние игнорируются)
function first(a, b) {
  return a;
}

console.log(first(1, 2, 3, 4)); // 1

// Пример 5: Меньше аргументов, чем параметров (остальные undefined)
function divide(a, b) {
  console.log("a =", a);
  console.log("b =", b);
  return a / b;
}

console.log(divide(10)); // a = 10, b = undefined, результат NaN

Важные моменты

  • Параметры — это локальные переменные функции
  • Изменение параметра внутри функции не влияет на исходную переменную
  • Параметры, для которых не передано значение, равны undefined
  • Лишние аргументы игнорируются
  • Используйте arguments для доступа ко всем аргументам (старый способ)

Объект arguments (устаревший способ):

function sum() {
  let total = 0;
  for (let i = 0; i < arguments.length; i++) {
    total += arguments[i];
  }
  return total;
}

console.log(sum(1, 2, 3));      // 6
console.log(sum(10, 20, 30, 40)); // 100

// Современный способ - см. Rest параметр ниже

Оператор Return

Return завершает выполнение функции и возвращает значение вызывающему коду. Если return не указан, функция возвращает undefined.

Синтаксис:

function functionName(parameters) {
  // код
  return value; // Возвращает значение и выходит из функции
  // код ниже не выполнится
}

const result = functionName(args);
console.log(result); // value

Примеры Return:

// Пример 1: Возврат значения
function multiply(a, b) {
  return a * b; // Возвращает результат
}

const result = multiply(5, 3);
console.log(result); // 15

// Пример 2: Return без значения (возвращает undefined)
function logMessage(message) {
  console.log(message);
  return; // или просто return без значения
}

const res = logMessage("Привет");
console.log(res); // undefined

// Пример 3: Ранний выход из функции
function checkNumber(num) {
  if (num < 0) {
    return "Число отрицательное";
  }
  if (num === 0) {
    return "Число равно нулю";
  }
  return "Число положительное";
}

console.log(checkNumber(-5));  // "Число отрицательное"
console.log(checkNumber(0));   // "Число равно нулю"
console.log(checkNumber(10));  // "Число положительное"

// Пример 4: Возврат объекта
function getUserInfo(name, age) {
  return {
    name: name,
    age: age,
    isAdult: age >= 18
  };
}

const user = getUserInfo("Alice", 25);
console.log(user); // { name: "Alice", age: 25, isAdult: true }

// Пример 5: Возврат функции
function createMultiplier(multiplier) {
  return function(x) {
    return x * multiplier;
  };
}

const double = createMultiplier(2);
console.log(double(5)); // 10

// Пример 6: Возврат массива
function getCoordinates() {
  return [10, 20, 30];
}

const [x, y, z] = getCoordinates();
console.log(x, y, z); // 10, 20, 30

Важные моменты о Return

  • Return прерывает выполнение функции — код после return не выполняется
  • Функция может иметь несколько return (в разных ветвях if/else)
  • Без return функция возвращает undefined
  • Можно возвращать любой тип: число, строку, объект, массив, функцию
  • Return на разных строках требует правильного синтаксиса

⚠️ Опасность: return на новой строке

// ❌ Плохо - возвращает undefined!
function getValue() {
  return
  {
    value: 42
  };
}

console.log(getValue()); // undefined

// ✅ Хорошо - возвращает объект
function getValue() {
  return {
    value: 42
  };
}

console.log(getValue()); // { value: 42 }

Параметры по умолчанию

Параметры по умолчанию — это значения, которые используются, если аргумент не был передан при вызове функции.

Синтаксис:

function greet(name = "Гость", greeting = "Привет") {
  console.log(greeting + ", " + name);
}

greet("Alice", "Hello");  // "Hello, Alice"
greet("Bob");             // "Привет, Bob"
greet();                  // "Привет, Гость"

Примеры параметров по умолчанию:

// Пример 1: Простые значения по умолчанию
function createUser(name = "Anonymous", role = "user") {
  return { name, role };
}

console.log(createUser("Alice", "admin"));  // { name: "Alice", role: "admin" }
console.log(createUser("Bob"));             // { name: "Bob", role: "user" }
console.log(createUser());                  // { name: "Anonymous", role: "user" }

// Пример 2: Вычисляемые значения по умолчанию
function calculatePrice(price, discount = price * 0.1) {
  return price - discount;
}

console.log(calculatePrice(100));      // 90 (100 - 10)
console.log(calculatePrice(100, 20));  // 80 (100 - 20)

// Пример 3: Параметры по умолчанию зависят от других параметров
function buildMessage(name, message = `Привет, ${name}!`) {
  return message;
}

console.log(buildMessage("Alice"));           // "Привет, Alice!"
console.log(buildMessage("Bob", "Hi, Bob!")); // "Hi, Bob!"

// Пример 4: Объект с параметрами по умолчанию
function configureServer(config = {}) {
  const { host = "localhost", port = 3000, ssl = false } = config;
  console.log(`${ssl ? "https" : "http"}://${host}:${port}`);
}

configureServer();                                    // "http://localhost:3000"
configureServer({ host: "example.com", ssl: true }); // "https://example.com:3000"

// Пример 5: Функция по умолчанию
function processData(data, callback = (x) => x) {
  return callback(data);
}

console.log(processData(5));              // 5
console.log(processData(5, (x) => x * 2)); // 10

Различие между undefined и отсутствием аргумента

function test(x = 100) {
  console.log(x);
}

test();          // 100 (параметр по умолчанию)
test(undefined); // 100 (undefined также использует значение по умолчанию!)
test(null);      // null (null не использует значение по умолчанию)
test(0);         // 0 (0 не использует значение по умолчанию)

Rest (...) и Spread (...) операторы

Rest оператор (...) собирает оставшиеся аргументы в массив. Spread оператор (...) раскрывает массив или объект в отдельные элементы.

Rest параметр

Собирает все оставшиеся аргументы в массив. Должен быть последним параметром.

Примеры Rest:

// Пример 1: Rest параметр
function sum(...numbers) {
  let total = 0;
  for (let num of numbers) {
    total += num;
  }
  return total;
}

console.log(sum(1, 2, 3));        // 6
console.log(sum(10, 20, 30, 40)); // 100

// Пример 2: Rest с другими параметрами
function buildString(separator, ...words) {
  return words.join(separator);
}

console.log(buildString(" ", "Hello", "World", "!"));      // "Hello World !"
console.log(buildString("-", "one", "two", "three"));      // "one-two-three"

// Пример 3: Rest для сбора некоторых параметров
function createList(title, ...items) {
  return {
    title,
    items,
    count: items.length
  };
}

const list = createList("Фрукты", "яблоко", "банан", "груша");
console.log(list);
// { title: "Фрукты", items: ["яблоко", "банан", "груша"], count: 3 }

Spread оператор

Раскрывает массив или объект в отдельные элементы.

Примеры Spread:

// Пример 1: Spread для распаковки аргументов
function add(a, b, c) {
  return a + b + c;
}

const numbers = [1, 2, 3];
console.log(add(...numbers)); // 6

// Пример 2: Spread для копирования массива
const arr1 = [1, 2, 3];
const arr2 = [...arr1]; // Копия массива
arr2.push(4);

console.log(arr1); // [1, 2, 3]
console.log(arr2); // [1, 2, 3, 4]

// Пример 3: Spread для объединения массивов
const first = [1, 2];
const second = [3, 4];
const combined = [...first, ...second]; // [1, 2, 3, 4]

// Пример 4: Spread для копирования объекта
const user = { name: "Alice", age: 25 };
const userCopy = { ...user }; // Копия объекта
userCopy.age = 30;

console.log(user.age);     // 25
console.log(userCopy.age); // 30

// Пример 5: Spread с добавлением свойств
const config = { host: "localhost", port: 3000 };
const fullConfig = { ...config, ssl: true, timeout: 5000 };

console.log(fullConfig);
// { host: "localhost", port: 3000, ssl: true, timeout: 5000 }

Сравнение Rest и Spread:

Rest (...) Spread (...)
Собирает элементы в массив Раскрывает элементы из массива
В параметрах функции При вызове функции или создании массива
function(...args) {} func(...array)
Справа от = Слева от =

Практические примеры

Пример 1: Функция с опциональными параметрами

function formatDate(date = new Date(), format = "DD/MM/YYYY") {
  // Простой пример форматирования
  const day = String(date.getDate()).padStart(2, "0");
  const month = String(date.getMonth() + 1).padStart(2, "0");
  const year = date.getFullYear();

  return format
    .replace("DD", day)
    .replace("MM", month)
    .replace("YYYY", year);
}

console.log(formatDate());                                // "31/12/2024"
console.log(formatDate(new Date(2025, 0, 15)));          // "15/01/2025"
console.log(formatDate(new Date(2025, 0, 15), "YYYY-MM-DD")); // "2025-01-15"

Пример 2: Функция с переменным числом аргументов

function average(...numbers) {
  if (numbers.length === 0) return 0;

  const sum = numbers.reduce((acc, num) => acc + num, 0);
  return sum / numbers.length;
}

console.log(average());           // 0
console.log(average(10, 20, 30)); // 20
console.log(average(5, 15));      // 10

Пример 3: Объединение rest и spread

function mergeObjects(target, ...sources) {
  return sources.reduce((result, source) => {
    return { ...result, ...source };
  }, target);
}

const base = { a: 1, b: 2 };
const obj1 = { b: 3, c: 4 };
const obj2 = { c: 5, d: 6 };

const merged = mergeObjects(base, obj1, obj2);
console.log(merged); // { a: 1, b: 3, c: 5, d: 6 }

Ловушки и рекомендации

⚠️ Основные ловушки

  • Return на новой строке может быть интерпретирован как завершение (используйте объекты со скобкой на той же строке)
  • Изменение параметров объекта влияет на исходный объект (передача по ссылке)
  • Параметры по умолчанию не применяются к undefined, но можно обойти это
  • Rest параметр должен быть последним в списке параметров
  • Spread в объектах перезаписывает свойства с одинаковыми ключами

✅ Лучшие практики

  • Используйте параметры по умолчанию вместо проверки if (!param)
  • Используйте rest параметры для функций с переменным числом аргументов
  • Используйте spread для копирования и объединения массивов/объектов
  • Копируйте объекты при передаче в функцию, если нужна неизменяемость
  • Документируйте параметры и возвращаемое значение функции
