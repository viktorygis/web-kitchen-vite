Scope и Hoisting в JavaScript

Scope (область видимости) определяет, где в коде можно получить доступ к переменной. Hoisting (поднятие) — это механизм, при котором объявления переменных и функций перемещаются в начало области видимости перед выполнением кода.

Основные концепции

  • Global Scope — переменные, доступные везде
  • Function Scope — переменные, видимые только внутри функции
  • Block Scope — переменные, видимые только в блоке (let, const)
  • Lexical Scope — вложенные функции имеют доступ к внешним переменным
  • Hoisting — объявления поднимаются в начало scope
  • Temporal Dead Zone (TDZ) — период до инициализации переменной

Global Scope: глобальная область видимости

Global Scope — это область видимости верхнего уровня. Переменные, объявленные в global scope, доступны везде в коде.

Примеры Global Scope:

// Глобальная переменная
let globalVar = "Я глобальная";

function test() {
  console.log(globalVar); // "Я глобальная" - доступна в функции
}

{
  console.log(globalVar); // "Я глобальная" - доступна в блоке
}

// Пример 2: Переменная без let/const/var (плохая практика)
// ❌ Плохо - создаёт глобальную переменную
function badFunction() {
  badGlobal = "Я глобальная, но объявлена неправильно";
}

// Пример 3: В браузере глобальные переменные становятся свойствами window
let browserGlobal = "значение";
console.log(window.browserGlobal); // "значение"

// Пример 4: Использование глобальных переменных в разных функциях
let counter = 0;

function increment() {
  counter++;
}

function decrement() {
  counter--;
}

function getCounter() {
  return counter;
}

increment();
increment();
console.log(getCounter()); // 2

⚠️ Избегайте глобальных переменных

  • Глобальные переменные загрязняют namespace
  • Сложно отследить, где переменная изменяется
  • Конфликты имён в больших проектах
  • Используйте локальные переменные и модули

Function Scope: область видимости функции

Function Scope — переменные, объявленные внутри функции, видны только в этой функции. При использовании var создаётся function scope.

Примеры Function Scope:

// Пример 1: var создаёт function scope
function myFunction() {
  var functionScoped = "видна только в функции";

  console.log(functionScoped); // "видна только в функции"
}

console.log(functionScoped); // ReferenceError: functionScoped is not defined

// Пример 2: Каждая функция имеет свой scope
function first() {
  var x = 1;
}

function second() {
  var x = 2;
  console.log(x); // 2
}

first();
second();
console.log(x); // ReferenceError

// Пример 3: Параметры функции видны только в функции
function greet(name) {
  console.log("Привет, " + name);
}

greet("Alice");
console.log(name); // ReferenceError

// Пример 4: var внутри блока всё ещё видна в функции
function varExample() {
  if (true) {
    var x = 10;
  }
  console.log(x); // 10 (var не создаёт block scope!)
}

varExample();

// Пример 5: Вложенные функции
function outer() {
  var outerVar = "внешняя";

  function inner() {
    var innerVar = "внутренняя";
    console.log(outerVar); // "внешняя" - видна!
    console.log(innerVar); // "внутренняя"
  }

  inner();
  console.log(innerVar); // ReferenceError
}

Block Scope: область видимости блока

Block Scope — переменные, объявленные с let и const, видны только в блоке (фигурные скобки). Это отличие от var, который создаёт function scope.

Примеры Block Scope:

// Пример 1: let создаёт block scope
{
  let blockScoped = "видна только в блоке";
  const also = "тоже видна только в блоке";

  console.log(blockScoped); // "видна только в блоке"
}

console.log(blockScoped); // ReferenceError

// Пример 2: Блоки в циклах
for (let i = 0; i < 3; i++) {
  console.log(i); // 0, 1, 2
}

console.log(i); // ReferenceError - i не видна снаружи

// Для сравнения с var:
for (var j = 0; j < 3; j++) {
  console.log(j); // 0, 1, 2
}

console.log(j); // 3 - j видна снаружи!

// Пример 3: let в условиях
if (true) {
  let x = 10;
  const y = 20;
  console.log(x, y); // 10, 20
}

console.log(x); // ReferenceError
console.log(y); // ReferenceError

// Пример 4: Каждая итерация цикла имеет свой scope (с let)
const callbacks = [];

for (let i = 0; i < 3; i++) {
  callbacks.push(() => console.log(i));
}

callbacks[0](); // 0
callbacks[1](); // 1
callbacks[2](); // 2

// С var результат был бы другим:
const callbacks2 = [];

for (var k = 0; k < 3; k++) {
  callbacks2.push(() => console.log(k));
}

callbacks2[0](); // 3
callbacks2[1](); // 3
callbacks2[2](); // 3

Lexical Scope: лексическая область видимости

Lexical Scope — это способность вложенных функций получать доступ к переменным из внешней функции. Определяется в момент написания кода, а не выполнения.

Примеры Lexical Scope:

// Пример 1: Вложенная функция видит внешние переменные
function outer() {
  let outerVar = "внешняя";

  function inner() {
    console.log(outerVar); // "внешняя"
  }

  inner();
}

outer();

// Пример 2: Цепочка scope
function level1() {
  let var1 = "уровень 1";

  function level2() {
    let var2 = "уровень 2";

    function level3() {
      let var3 = "уровень 3";

      console.log(var1); // "уровень 1"
      console.log(var2); // "уровень 2"
      console.log(var3); // "уровень 3"
    }

    level3();
  }

  level2();
}

level1();

// Пример 3: Изменение внешней переменной
function counter() {
  let count = 0;

  return function() {
    count++;
    console.log(count);
  };
}

const increment = counter();
increment(); // 1
increment(); // 2
increment(); // 3

// Пример 4: Параметры функции видны во вложенных функциях
function outer2(param) {
  function inner2() {
    console.log(param); // видна!
  }

  inner2();
}

outer2("значение"); // "значение"

// Пример 5: Shadowing - переопределение переменной
let x = "глобальная";

function test() {
  let x = "локальная"; // Shadowing!

  function inner() {
    console.log(x); // "локальная" (используется ближайшая переменная)
  }

  inner();
}

test();
console.log(x); // "глобальная"

Hoisting: поднятие объявлений

Hoisting — это процесс, при котором объявления переменных и функций перемещаются в начало scope перед выполнением кода. Однако инициализация остаётся на месте.

Hoisting с var:

// Код, как мы его пишем:
console.log(x);
var x = 5;
console.log(x);

// Как JavaScript интерпретирует (примерно):
var x;           // Объявление поднято
console.log(x);  // undefined (объявлена, но не инициализирована)
x = 5;           // Инициализация на месте
console.log(x);  // 5

// Пример 2: Функции поднимаются полностью
sayHello(); // "Привет!" - работает!

function sayHello() {
  console.log("Привет!");
}

// Пример 3: Переменные после return
function test() {
  console.log(x); // undefined

  if (false) {
    var x = 10;
  }

  console.log(x); // undefined
}

test();

Hoisting с let и const:

// Пример 1: let не поднимается в начало выполнения
console.log(y); // ReferenceError: Cannot access 'y' before initialization

let y = 5;

// Пример 2: Temporal Dead Zone (TDZ)
function testTDZ() {
  // TDZ начинается здесь
  console.log(z); // ReferenceError

  let z = 10; // TDZ заканчивается здесь
}

// Пример 3: const также имеет TDZ
console.log(const1); // ReferenceError

const const1 = "значение";

Различия в Hoisting

var let / const Function
Поднимается и инициализируется как undefined Поднимается, но не инициализируется (TDZ) Полностью поднимается
Можно использовать до объявления (будет undefined) ReferenceError если использовать до объявления Можно вызывать до объявления
Function scope Block scope Не применимо

Var, Let, Const: разница и выбор

Var, Let и Const — это три способа объявления переменных в JavaScript. Современный стандарт рекомендует использовать let и const.

Характеристика var let const
Scope Function scope Block scope Block scope
Hoisting Поднимается (undefined) Поднимается (TDZ) Поднимается (TDZ)
Переопределение Да Да Нет
Изменение значения Да Да Нет
Глобальная переменная (browser) window.x нет нет
Рекомендация ❌ Не используйте ✅ По умолчанию ✅ Для констант

Примеры сравнения:

// Пример 1: Block scope
{
  var a = 1;
  let b = 2;
  const c = 3;
}

console.log(a);     // 1 (видна)
console.log(b);     // ReferenceError
console.log(c);     // ReferenceError

// Пример 2: Переопределение
var x = 1;
var x = 2; // OK

let y = 1;
let y = 2; // SyntaxError

const z = 1;
const z = 2; // SyntaxError

// Пример 3: Изменение значения
var v1 = 1;
v1 = 2; // OK

let v2 = 1;
v2 = 2; // OK

const v3 = 1;
v3 = 2; // TypeError

// Пример 4: const с объектами
const obj = { name: "Alice" };
obj.name = "Bob"; // OK - изменение свойства
// obj = {}; // TypeError - переопределение переменной

// Пример 5: Рекомендуемый стиль
const name = "Alice";        // Константа
let age = 25;                // Переменная
let email = "alice@mail.com";

// Избегайте:
// var oldStyle = "не используйте";

✅ Рекомендации по выбору

  • Используйте const по умолчанию для всего
  • Используйте let, если переменная будет изменяться
  • Избегайте var — это устаревший синтаксис
  • Const не запрещает изменение свойств объекта, только переопределение переменной

Практические примеры

Пример 1: Closure с сохранением состояния

function createBank(initialBalance) {
  let balance = initialBalance; // Сохраняется в closure

  return {
    deposit: function(amount) {
      balance += amount;
      return balance;
    },
    withdraw: function(amount) {
      balance -= amount;
      return balance;
    },
    getBalance: function() {
      return balance;
    }
  };
}

const myBank = createBank(1000);
console.log(myBank.deposit(500));   // 1500
console.log(myBank.withdraw(200));  // 1300
console.log(myBank.getBalance());   // 1300

Пример 2: Проблема с var в цикле

// Проблема с var
const callbacks1 = [];
for (var i = 0; i < 3; i++) {
  callbacks1.push(function() {
    console.log(i);
  });
}
callbacks1[0](); // 3 (неожиданно!)
callbacks1[1](); // 3
callbacks1[2](); // 3

// Решение 1: Использовать let
const callbacks2 = [];
for (let i = 0; i < 3; i++) {
  callbacks2.push(function() {
    console.log(i);
  });
}
callbacks2[0](); // 0
callbacks2[1](); // 1
callbacks2[2](); // 2

// Решение 2: IIFE с var (старый способ)
const callbacks3 = [];
for (var i = 0; i < 3; i++) {
  callbacks3.push((function(j) {
    return function() {
      console.log(j);
    };
  })(i));
}
callbacks3[0](); // 0
callbacks3[1](); // 1
callbacks3[2](); // 2

Пример 3: Hoisting функций

// Функции поднимаются полностью
console.log(add(2, 3)); // 5

function add(a, b) {
  return a + b;
}

// Function expression НЕ поднимается
console.log(multiply(2, 3)); // TypeError: multiply is not a function

var multiply = function(a, b) {
  return a * b;
};

// С let/const ещё хуже (TDZ)
console.log(divide(10, 2)); // ReferenceError

const divide = function(a, b) {
  return a / b;
};

Ловушки и рекомендации

⚠️ Основные ловушки

  • Hoisting var — переменная доступна, но равна undefined
  • Function scope var — может привести к непредсказуемому поведению в циклах
  • Shadowing — переопределение переменной во вложенном scope
  • Забывчивость const/let — ReferenceError вместо undefined
  • TDZ (Temporal Dead Zone) — переменная поднята, но недоступна
  • Глобальные переменные — загрязняют namespace

✅ Лучшие практики

  • Используйте const и let, забудьте про var
  • Объявляйте переменные в начале блока, где они нужны
  • Избегайте глобальных переменных — используйте модули
  • Понимайте разницу между block scope и function scope
  • Используйте closures для инкапсуляции данных
  • Будьте осторожны с hoisting — особенно при использовании var
