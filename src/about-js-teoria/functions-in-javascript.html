Функции в JavaScript

Функции — это основа программирования. Это блоки кода, которые выполняют конкретную задачу, и их можно запускать много раз. Они помогают:

  • Повторно использовать код : написал один раз — вызывай когда нужно.
  • Организовывать программу : дели большой код на небольшие понятные части.
  • Делать код читаемым : хорошая функция = понятная мини-команда.

Что такое функция простыми словами?

Представь рецепт. Ты записал шаги приготовления блюда — это и есть функция. Рецепт можно использовать снова и снова. Чтобы «приготовить блюдо» — нужно выполнить рецепт. Чтобы выполнить код функции — её нужно вызвать.

Как создать функцию?

Используй ключевое слово function , за ним — имя функции, круглые скобки () , затем фигурные { } с кодом внутри.

  1. Напиши function
  2. Придумай понятное имя (например, greet )
  3. Добавь круглые скобки (там позже будут параметры)
  4. В фигурных скобках — что должна делать функция

Пример:

function greet() {
  console.log("Привет, мир!");
}

Совет: Имя должно описывать действие: printReport , loadUser , calcTotal .

Как вызвать функцию?

Чтобы выполнить функцию — напиши её имя и добавь круглые скобки: greet();

Пример вызова:

function greet() {
  console.log("Привет!");
}

greet(); // Вызов функции
greet(); // Можно снова
greet(); // И ещё раз

Что происходит при вызове?
JS находит функцию → выполняет её код → возвращается к месту вызова.

Параметры и аргументы

Параметры — «переменные-заглушки» внутри круглых скобок при объявлении функции.
Аргументы — реальные значения, которые передаём при вызове.

Простой пример:

function greet(name) {        // name — параметр
  console.log("Привет, " + name + "!");
}

greet("Кодик");               // "Кодик" — аргумент
// Привет, Кодик!

Несколько параметров:

function introduce(name, age) {
  console.log("Меня зовут " + name + ", мне " + age + " лет.");
}

introduce("Лена", 25);
// Меня зовут Лена, мне 25 лет.

Советы:
Имена параметров — понятные. Можно передавать любые типы: строки, числа, объекты, массивы, функции.

Возврат значения

Функции в JavaScript могут не только выполнять код, но и возвращать результат . Для этого используется ключевое слово return . Как только интерпретатор встречает return — выполнение функции прекращается, и указанное значение передаётся «наружу».

Что можно вернуть? Любое значение: число, строку, булево, массив, объект, null , undefined , другую функцию.

Базовый пример

function sum(a, b) {
  return a + b; // Возвращаем сумму
}

let result = sum(5, 10);         // Сохраняем результат
console.log("Сумма:", result);   // Сумма: 15

// Пояснение:
// 1. sum получает a и b
// 2. вычисляет a + b
// 3. возвращает значение в место вызова
// 4. мы сохраняем его в result и используем дальше

С return

function multiply(a, b) {
  return a * b;
}

let m = multiply(3, 4);
console.log(m); // 12

Без return

function logMultiply(a, b) {
  console.log(a * b);
}

let v = logMultiply(3, 4); // выведет 12
console.log(v);            // undefined

Возврат объекта / функции

function makeUser(name) {
  return { name, created: Date.now() };
}

function makeAdder(x) {
  return function(y) {
    return x + y;
  };
}

const user = makeUser("Аня");
console.log(user.name); // "Аня"

const add5 = makeAdder(5);
console.log(add5(3)); // 8
Почему это полезно?
  • Можно сохранить результат и использовать позже
  • Код становится модульным и переиспользуемым
  • Проще тестировать (функция возвращает «чистый» результат)
  • Легче комбинировать функции между собой

Остановка выполнения после return

function check(n) {
  if (n < 0) {
    return "Отрицательное";
  }
  console.log("Это сообщение выполнится только если n >= 0");
  return "Неотрицательное";
}
Опасность автоматической вставки точки с запятой (ASI):
Не переносите выражение на новую строку сразу после return . 
function bad() {
  return
  5; // интерпретатор вставит ; после return → вернёт undefined
}

function good() {
  return 5; // правильно
}

Мини-шпаргалка

// Возврат простого значения
return 42;

// Возврат выражения
return a * (b + c);

// Возврат объекта
return { ok: true };

// Возврат ничего (явно)
return;          // undefined

// Без return (неявно)
function f() {}
f(); // undefined

Область видимости (scope)

Область видимости — это зона, в которой переменная доступна. Понимание различий помогает избежать ошибок и конфликтов имён.

👉 Локальные переменные

  • Создаются внутри функции
  • Видны только внутри этой функции
  • Недоступны снаружи
function test() {
  let localVar = "Я локальная!";
  console.log(localVar); // Видна тут
}

test();
console.log(localVar); // Ошибка: localVar is not defined

🌍 Глобальные переменные

  • Создаются вне функций (в верхнем уровне файла/скрипта)
  • Доступны везде (если не в модульной изоляции)
  • Видны внутри функций
let globalVar = "Я глобальная!";

function show() {
  console.log(globalVar); // Видна
}

show();
console.log(globalVar); // Тоже видна

Пример: локальная + глобальная вместе

let globalVar = "Я глобальная!"; // Глобальная переменная

function test() {
  let localVar = "Я локальная!"; // Локальная
  console.log(globalVar); // Видна
  console.log(localVar);  // Видна
}

test();

console.log(globalVar); // Видна
console.log(localVar);  // Ошибка! localVar не существует здесь

Изменение глобальной внутри функции

let count = 0;

function inc() {
  count++; // Меняем глобальную
}

inc();
inc();
console.log(count); // 2

«Затенение» (shadowing)

let user = "Вася";

function demo() {
  let user = "Петя"; // Новая локальная, не изменяет глобальную
  console.log(user); // Петя (локальная)
}

demo();
console.log(user); // Вася (глобальная)

Функциональная область (var) vs блочная (let/const)

if (true) {
  var a = 1;      // function-scoped (утечёт наружу)
  let b = 2;      // block-scoped
  const c = 3;    // block-scoped
}
console.log(a); // 1
console.log(b); // Ошибка
console.log(c); // Ошибка
Советы:
  • Минимизируй количество глобальных переменных
  • Используй let / const вместо var для предсказуемой области видимости
  • Давай переменным понятные имена в своей области
  • Избегай неявного создания глобальных (без let / const / var )
Опасность: Случайно пропущенное ключевое слово может создать глобальную переменную (в нестрогом режиме): 
function oops() {
  leaked = 42; // ❌ создаст глобальную leaked (если нет "use strict")
}
В строгом режиме ( "use strict" ) это вызовет ошибку — и это хорошо.

Мини-шпаргалка

// Локальная
function f() {
  let x = 10;
}

// Глобальная (в файле)
let g = 5;

// Затенение
let value = 1;
function show() {
  let value = 2;
  console.log(value); // 2
}
show();
console.log(value);   // 1

Параметры по умолчанию

Если аргумент не передан — можно задать значение по умолчанию.

ES6 способ:

function greet(name = "Гость") {
  console.log("Привет, " + name + "!");
}

greet();        // Привет, Гость!
greet("Аня");   // Привет, Аня!

Старый способ:

function greet(name) {
  name = name || "Гость";
  console.log("Привет, " + name + "!");
}

Лучше использовать современный синтаксис (param = value) .

Именование функций

Имя функции — глагол + уточнение. Оно должно отвечать на вопрос: «Что делает функция?»

Префикс Смысл Примеры Что возвращает
get Получить данные getUser, getAge Значение
set / update Изменить данные setTheme, updateUser Обычно void
calc Вычислить calcTotal, calcAverage Число
create Создать что-то createUser, createButton Объект
is / has / can Проверка isReady, hasError Boolean

Неудачные: doStuff() , handle() , process() — слишком общие.
Хорошие: loadProfile() , filterActiveUsers() , formatDate() .

Лучшие практики

Плохой пример (делает слишком много):

function processUser(user) {
  if (!user.name || !user.email) {
    alert("Ошибка");
    return;
  }
  saveToDatabase(user);
  sendWelcomeEmail(user.email);
  updateUserList();
}

Лучше разбить:

function isValidUser(user) {
  return Boolean(user.name && user.email);
}

function saveUser(user) {
  saveToDatabase(user);
}

function notifyUser(user) {
  sendWelcomeEmail(user.email);
}

function registerUser(user) {
  if (!isValidUser(user)) {
    throw new Error("Неполные данные");
  }
  saveUser(user);
  notifyUser(user);
  updateUserList();
}

Почему так лучше?
Проще тестировать, легче читать, ниже связанность, выше переиспользование.

Функции как «живые комментарии»:

function showPrimes(limit) {
  for (let n = 2; n < limit; n++) {
    if (isPrime(n)) {
      console.log(n);
    }
  }
}

function isPrime(num) {
  for (let d = 2; d < num; d++) {
    if (num % d === 0) return false;
  }
  return true;
}

Сводка

Ключевые идеи:

  • Функция — переиспользуемый блок кода
  • Создаётся через function имя()
  • Вызывается через имя()
  • Параметры принимают данные
  • return возвращает результат
  • Локальные переменные живут внутри функции
  • Используйте понятные имена
  • Одна функция = одно действие

Мини-шпаргалка:

// Объявление
function add(a, b) {
  return a + b;
}

// Вызов
add(2, 3); // 5

// Параметр по умолчанию
function greet(name = "Гость") {
  console.log("Привет, " + name);
}

// Проверка
function isAdult(age) {
  return age >= 18;
}
